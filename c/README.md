
# Prerequisites
- libcurl4-openssl-dev
- libssl-dev
- cmake
- make
- gcc


# Installation
1. make build
2. sudo make install

# Docker Example
1. docker build . -t avh-c-example:latest
2. docker run -it avh-c-example https://app.avh.arm.com/api <ApiToken>
